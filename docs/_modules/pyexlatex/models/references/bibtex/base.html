    def __str__(self):
        str_ = bibtex_str(
            self.item_type,
            self.item_accessor,
            self.fields
        )

        return str_

    @property
    def fields(self) -> Dict[str, str]:
        fields_dict = {arg: getattr(self, arg) for arg in self.required_attrs}

        for attr in self.optional_attrs:
            self_attr = getattr(self, attr)
            if self_attr:
                fields_dict.update({attr: self_attr})

        return fields_dict
